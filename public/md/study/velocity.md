    const tl = gsap.timeline({repeat:-1});
    this.tl.push(tl);

    const Dtext = document.querySelectorAll('.D-text');

    tl.fromTo(DLaptop[1], {
    autoAlpha:0,
    }, {
    autoAlpha:1,
    duration:1
    },'>5');
    tl.fromTo(DLaptop[2], {
    autoAlpha:0,
    }, {
    autoAlpha:1,
    duration:1
    },'>5') ;

    tl.fromTo([DLaptop[1], DLaptop[2]], {
    autoAlpha:1,
    }, {
    autoAlpha:0,
    duration:1
    },'>5');



    tl.fromTo(Dtext[1], {
    autoAlpha:0,
    y:'10%'
    }, {
    autoAlpha:1,
    y:0,
    duration:1
    },'5');
    tl.fromTo(Dtext[0], {
    autoAlpha:1,
    y:0
    }, {
    autoAlpha:0,
    y:'-10%',
    duration:1
    },'<');

    tl.fromTo(Dtext[2], {
    autoAlpha:0,
    y:'10%'
    }, {
    autoAlpha:1,
    y:0,
    duration:1
    },'>5');

    tl.fromTo(Dtext[1], {
    autoAlpha:1,
    y:0
    }, {
    autoAlpha:0,
    y:'-10%',
    duration:1
    },'<');

    y:0,

    },'>5');

    },'<');
